\section{Introduction, motivation and main definitions}

In this paper all topological spaces are assumed to be Hausdorff.

A \emph{topological semigroup} is a Hausdorff topological space together with a continuous semigroup operation . If $S$ is a semigroup and $\tau$ is a topology on $S$ such that $(S,\tau)$ is a topological semigroup, then we shall call $\tau$ \emph{semigroup topology} on $S$. A \emph{semitopological semigroup} is a Hausdorff topological space together with a separately continuous semigroup operation. Topological semigroup $(S,\tau)$ is said to be \emph{minimal} if no semigroup topology on $S$ is strictly contained in $\tau$. If $(S,\tau)$ is minimal topological semigroup, then $\tau$ is called \emph{minimal semigroup topology}.

The concept of minimal topological groups was introduced independently in the early 1970’s by Doitchinov \cite{Doitchinov} and Stephenson \cite{Stephenson}. Both authors were motivated by the theory of minimal topological spaces, which was well understood at that time. More than 20 years earlier L. Nachbin \cite{Nachbin} had studied minimality in the context of division rings, and B. Banaschewski \cite{Banaschewski} investigated minimality in the more general setting of topological algebras.

Let $\lambda$ be a nonempty set. By $\B$ we denote the set $\lambda\times\lambda\cup\{0\}$ endowed with the following semigroup operation:
\begin{equation*}
  (\alpha,\beta)\cdot (\gamma,\delta)=\left\{
     \begin{array}{ll}
       (\alpha,\delta), & \hbox{$\beta=\gamma$;} \\
       0, & \hbox{$\beta\neq\gamma$}
     \end{array}
   \right.
\end{equation*}

and $(\alpha,\beta)\cdot 0=0\cdot(\alpha,\beta)=0\cdot 0=0$, for each $\alpha,\beta,\gamma,\delta\in\ab$. The semigroup $\B$ is called the \emph{semigroup of $\lambda\times\lambda$-matrix units}. The semitopological and topological semigroup of matrix units was investigated in \cite{Pavlyk}.

A \emph{directed graph} (or just \emph{digraph}) $D$ consists of a nonempty set $V(D)$ of elements called \emph{vertices} and a set $A(D)$ of ordered pairs of vertices called \emph{arcs}. We call $V(D)$ the \emph{vertex set} and $A(D)$ the \emph{arc set} of $D$. The \emph{order} (\emph{size}) of $D$ is the cardinality of the vertex (arc) set of $D$. For an arc $(u,v)$ the first vertex $u$ is its \emph{tail} and the second vertex $v$ is its \emph{head}. If a tail and a head of arc coincide, then this arc is called a \emph{loop}. The head and tail of an arc are its \emph{end-vertices}. A vertex $v$ is a \emph{source}(\emph{source}) if $v$ is not a head(tail) for any arc. A digraph $H$ is a \emph{subdigraph} of a digraph $D$ if $V(H)\subseteq V(D)$, $A(H)\subseteq A(D)$ and every arc in $A(H)$ has both end-vertices in $V(H)$. If every arc of $A(D)$ with both end-vertices in $V(H)$ is in $A(H)$, we say that $H$ is induced by $X=V(H)$ and call $H$ an induced subdigraph of $D$.

A \emph{walk} in $D$ is an alternating sequence $W=x_{1}a_{1}x_{2}a_{2}x_{3}\ldots x_{k-1}a_{k-1}x_{k}$ of vertices $x_{i}$ and arcs $a_{j}$ from $D$ such that the tail of $a_{i}$ is $x_{i}$ and the head of $a_{i}$ is $x_{i+1}$ for every $i=1,2,\ldots,k-1$. The \emph{length} of a walk is the number of its arcs. When the arcs of $W$ are defined from the context or simply unimportant, we will denote $W$ by $x_{1}x_{2}\ldots x_{k}$. If the vertices of $W$ are distinct, $W$ is a \emph{path}. If the vertices $x_{1},x_{2},\ldots,x_{k-1}$ are distinct and $x_{1}=x_{k}$, $W$ is a \emph{cycle}. A walk (path, cycle) $W$ is a \emph{Hamilton} (or \emph{hamiltonian}) walk (path, cycle) if $W$ contains all vertices of $D$.

Let $\{D_{i}\}_{i\in I}$ be a family of digraphs. The digraph $(\bigsqcup\limits_{i\in I}V(D_{i}),\bigsqcup\limits_{i\in I}A(D_{i}))$ is called \emph{disjoint union} of this family and denoted by $\bigoplus\limits_{i\in I}D_{i}$. If $D$ is a digraph and $\mathcal{R}$ is an equivalence relation on $V(D)$. Then the \emph{quotient digraph} $D/\mathcal{R}$ has vertex set $V/\mathcal{R}$ and arc set $\{([a]_{\mathcal{R}},[b]_{\mathcal{R}})\mid (a,b)\in A(D)\}$.

\section{Compositional families}

If $(\B,\tau)$ is a semitopological semigroup, then any nonzero element of $\B$ is an isolated point of $(\B,\tau)$ \cite[Lemma 2]{Pavlyk}. Therefore the following lemma is true.

\begin{englemma}\label{lemma1}
Let $(\B,\tau)$ be a topological semigroup and $A$ be a closed subset $A$ of $(\B,\tau)$ which doesn't contain the zero $0$. Then any subset of $A$ is closed.
\end{englemma}

For $A\subseteq\B\setminus\{0\}$ and $\alpha,\beta\in\ab$ we denote
\begin{align*}
    _\alpha A_\beta&=\{(\beta,\gamma)\mid(\alpha,\gamma)\in A\};\\
    _\beta^\alpha A&=\{(\gamma,\beta)\mid(\gamma,\alpha)\in A\};\\
    \prf(A)&=\{\alpha\mid(\alpha,\beta)\in A\};\\
    \prs(A)&=\{\beta\mid(\alpha,\beta)\in A\}.\\
\end{align*}

\begin{englemma}\label{lemma2}
Let $\tau$ be a topology on $\B$ and any nonzero element of $\B$ is isolated in $(\B,\tau)$. The semigroup operation is continuous on $(\B\times\B)\setminus\{(0,0)\}$ if and only if the sets $_\alpha A_\beta$ and $_\beta^\alpha A$ are closed for all $\alpha,\beta\in\ab$ and any closed subset $A$ of $(\B,\tau)$ which doesn't contain the zero $0$.
\end{englemma}
\begin{engproof}
($\Rightarrow$) Let $A$ be a closed subset of $(\B,\tau)$ which doesn't contain the zero $0$. By the continuity of operation, the maps $\lambda_{(\alpha,\beta)}:\B\rightarrow \B$ and $\rho_{(\beta,\alpha)}:\B\rightarrow \B$ defined by the formulas $\lambda_{(\alpha,\beta)}(x)=(\alpha,\beta)\cdot x$ and $\rho_{(\beta,\alpha)}(x)=x\cdot(\beta,\alpha)$ are continuous. Therefore the sets
\begin{equation*}
  _\alpha A_\beta=(\lambda_{(\alpha,\beta)})^{-1}(A)
\end{equation*}
and
\begin{equation*}
  _\beta^\alpha A=(\rho_{(\beta,\alpha)})^{-1}(A)
\end{equation*}
are closed in the topological space $(\B,\tau)$.

($\Leftarrow$) Since every nonzero point of $\B$ is isolated, we check the continuity of semigroup operation only in the cases of $(\alpha,\beta)\cdot 0$ and $0\cdot(\alpha,\beta)$. Let $U$ be a neighborhood of the zero $0$ and $A=\B\setminus U$. Then the sets $_\alpha A_\beta$ and $_\beta^\alpha A$ are closed. Denote by $V$ and $W$ the neighborhoods of zero $\B\setminus {}_\alpha A_\beta$ and $\B\setminus {}_\beta^\alpha A$, respectively. Simple calculations imply that $\{(\alpha,\beta)\}\cdot V\subseteq U$ and $W\cdot\{(\alpha,\beta)\}\subseteq U$.
\end{engproof}

Lemma \ref{lemma2} implies the following corollary.

\begin{engcorollary}
Let $\tau$ be a topology on $\B$. Then $(\B,\tau)$ is a semitopological semigroup if and only if the sets $_\alpha A_\beta$ and $_\beta^\alpha A$ are closed for all $\alpha,\beta\in\ab$ and any closed subset $A$ of $(\B,\tau)$ such that $0\notin A$.
\end{engcorollary}

\begin{englemma}\label{point_get_out}
Let $A,B\subseteq\B$ and $(\alpha,\beta)\in\B$. The element $(\alpha,\beta)\notin A\cdot B$ if and only if $\prs(_\alpha A_\alpha)\cap\prf(_\beta^\beta B)=\varnothing$.
\end{englemma}
\begin{engproof}
($\Rightarrow$) Suppose that $(\alpha,\beta)\notin A\cdot B$ and there exists $\gamma\in\prs(_\alpha A_\alpha)\cap\prf(_\beta^\beta B)$. Then $(\alpha,\gamma)\in A$ and $(\gamma,\beta)\in B$. Therefore $(\alpha,\gamma)\cdot (\gamma,\beta)=(\alpha,\beta)\in A\cdot B$, a contradiction.

($\Leftarrow$) Suppose that $\prs(_\alpha A_\alpha)\cap\prf(_\beta^\beta B)=\varnothing$ and $(\alpha,\beta)\in A\cdot B$. Then there exist $\gamma\in\ab$ such that $(\alpha,\gamma)\in A$ and $(\gamma,\beta)\in B$. Hence $\gamma\in\prs(_\alpha A_\alpha)\cap\prf(_\beta^\beta B)$, a contradiction.
\end{engproof}

For any nonempty subsets $A$ and $B$ of $\ab$ we shall call $A\times B$ by a \emph{rectangle} of $\B$.

\begin{engdefinition}
A nonempty family $\mathcal{F}$ of rectangles of $\B$ is called compositional if for $A\times B\in \mathcal{F}$ there exists $C\subset\ab$ such that $A\times (\ab\setminus C)\in \mathcal{F}$ and $C\times B\in \mathcal{F}$.
\end{engdefinition}

\begin{englemma}\label{poligon_get_out}
If $\tau$ is a semigroup topology on $\B$, then the family of all closed rectangles of $(\B,\tau)$ is compositional.
\end{englemma}
\begin{engproof}
Since every point of $(\B,\tau)$ is a closed subset, the family of all closed rectangles of $\B$ is not empty. Let $A\times B$ be a closed subset of $(\B,\tau)$. Then $\B\setminus(A\times B)$ is a neighborhood of the zero $0$. Since the semigroup operation is continuous at $(0,0)$ there exist neighborhoods of zero $U$ and $V$ such that $U\cdot V\subseteq \B\setminus(A\times B)$. Lemma \ref{point_get_out} implies that for each $\alpha\in A$ the neighborhood $V$ doesn't contain the set $\prs(_\alpha U_\alpha)\times B$. Therefore, the neighborhood $V$ doesn't contain the set $\left(\bigcup\limits_{\alpha\in A}\prs(_\alpha U_\alpha)\right)\times B$. Denote the set $\bigcup\limits_{\alpha\in A}\prs(_\alpha U_\alpha)$ by $C$.

Suppose that $C\neq\ab$. By De Morgan's laws, $\ab\setminus C=\bigcap\limits_{\alpha\in A}\ab\setminus\prs(_\alpha U_\alpha)$. The neighborhood $U$ doesn't contain the set $A\times(\ab\setminus C)$. Lemma \ref{lemma1} implies that the sets $A\times (\ab\setminus C)$ and $C\times B$ are closed.

If $C=\ab$, then by Lemma \ref{lemma1} the set $(\ab\setminus\{\varphi\})\times B$ for some $\varphi\in\ab$ is closed. Lemma \ref{lemma2} implies that set $A\times\{\varphi\}$ is closed.

Hence the family of all closed rectangles of $(\B,\tau)$ is compositional.
\end{engproof}

Let $\mathcal{F}$ be compositional family. Denote
\begin{equation*}
\mathcal{C}=\mathcal{F}\cup\{_\alpha A_\beta,_\beta^\alpha A\mid A\in \mathcal{F} \text{ and }\alpha,\beta\in\ab\}\cup\{\{(\alpha,\beta)\}\mid (\alpha,\beta)\in B_{\lambda}\setminus\{0\}\}
\end{equation*}
 and
\begin{equation*}
P_{\mathcal{F}}=\{\B\setminus B\mid B\in \mathcal{C}\}\cup\{\{(\alpha,\beta)\}\mid (\alpha,\beta)\in B_{\lambda}\setminus\{0\}\}\cup\{\varnothing\}.
\end{equation*}

\begin{engproposition}\label{proposition2}
For every compositional family $\mathcal{F}$ the topology generated by the subbase $P_{\mathcal{F}}$ is the smallest semigroup topology on $\B$ such that elements of $\mathcal{F}$ are closed.
\end{engproposition}
\begin{engproof}
Let $\tau$ be the topology generated by $P_{\mathcal{F}}$. Since each nonzero point in $(\B,\tau)$ is clopen subset, the topological space $(\B,\tau)$ is Hausdorff.

First we shall show that topology generated by the subbase $P_{\mathcal{F}}$ is semigroup. For any $A\in\mathcal{C}$ the sets ${}_\alpha A_\beta$ and ${}_\beta^\alpha A$ are elements of $\mathcal{C}$. Since
\begin{align*}
  _\alpha(\bigcup\limits_{i\in I} A_{i})_\beta&=\bigcup\limits_{\iota\in I}{}_\alpha(A_{i})_\beta, \\
  _\beta^\alpha(\bigcup\limits_{i\in I} A_{i})&=\bigcup\limits_{\iota\in I}{}_\beta^\alpha(A_{i}), \\
  _\alpha(\bigcap\limits_{i\in I} A_{i})_\beta&=\bigcap\limits_{\iota\in I}{}_\alpha(A_{i})_\beta, \\
  _\beta^\alpha(\bigcap\limits_{i\in I} A_{i})&=\bigcap\limits_{\iota\in I}{}_\beta^\alpha(A_{i}), \\
\end{align*}
for arbitrary family $\{A_i\}_{i\in I}$ subsets of $\B\setminus\{0\}$, by Lemma \ref{lemma2}, the semigroup operation is continuous on $(\B\times \B)\setminus\{(0,0)\}$.

The continuity of the operation in the point $(0,0)$ can be verify only for elements of the subbase. Let $U$ be a neighborhood of $0$ such that $U\in P_{\mathcal{F}}$. Consider possible cases.
\begin{enumerate}
  \item If $\B\setminus U=A\times B\in \mathcal{F}$, then there exists $C\subset\ab$ such that $A\times(\ab\setminus C)$ and $C\times B$ are closed subsets of $(\B,\tau)$. Thus $\B\setminus(A\times(\ab\setminus C))$ and $\B\setminus(C\times B)$ are neighborhoods of the zero $0$ and
      \begin{equation*}
        (\B\setminus(A\times(\ab\setminus C)))\cdot(\B\setminus(C\times B))\subseteq\B\setminus(A\times B)=U.
      \end{equation*}
  \item If $\B\setminus U={}_\alpha(A\times B)_\beta=\{\beta\}\times B$ for some $\alpha\in A,\beta\in\ab$ and $A\times B\in \mathcal{F}$, then there exists $C\subset\ab$ such that the sets $A\times(\ab\setminus C)$ and $C\times B$ are closed in $(\B,\tau)$. The set $_\alpha(A\times(\ab\setminus C))_\beta=\{\beta\}\times(\ab\setminus C)$ is closed in $(\B,\tau)$. Hence $\B\setminus (\{\beta\}\times(\ab\setminus C))$ and $\B\setminus(C\times B)$ are neighborhoods of the zero $0$ and
      \begin{equation*}
        (\B\setminus (\{\beta\}\times(\ab\setminus C)))\cdot(\B\setminus(C\times B))\subseteq\B\setminus(\{\beta\}\times B)=U.
      \end{equation*}

  \item In the case $\B\setminus U={}_\beta^\alpha(A\times B)=A\times\{\beta\}$ the proof of the continuity of the semigroup operation is similar.
  
  \item Suppose that $\B\setminus U=\{(\alpha,\beta)\}$ for some $(\alpha,\beta)\in\B\setminus\{0\}$. Since family $\mathcal{C}$ is not empty, there exists $A\times B\in\mathcal{C}$. Then there exists $C\subset\ab$ such that the sets $A\times(\ab\setminus C)$ and $C\times B$ are closed in $(\B,\tau)$. Fix any $\varphi\in A$ and $\psi\in B$. Therefore the sets ${}_{\varphi} (A\times(\ab\setminus C))_{\alpha}$ and ${}_{\beta}^{\psi}(C\times B)$ are closed in $(\B,\tau)$. Thus $\B\setminus({}_{\varphi} (A\times(\ab\setminus C))_{\alpha})$ and $\B\setminus({}_{\beta}^{\psi}(C\times B))$ are neighborhoods of the zero $0$ and
      \begin{equation*}
        (\B\setminus({}_{\varphi} (A\times(\ab\setminus C))_{\alpha}))\cdot(\B\setminus({}_{\beta}^{\psi}(C\times B)))\subseteq\B\setminus\{(\alpha,\beta)\}=U.
      \end{equation*}
\end{enumerate}

Let $\tau$ be a semigroup topology on $\B$ such that elements of $\mathcal{F}$ are closed in the topological space $(\B,\tau)$. Then all nonzero points are closed. By Lemma \ref{lemma2}, elements of $\{_\alpha(A\times B)_\beta,_\beta^\alpha(A\times B)\mid A\in \mathcal{F} \text{ and }\alpha,\beta\in\ab\}$ are closed and, hence their complements are open. Since every nonzero point of $(\B,\tau)$ is isolated, elements of $P_{\mathcal{F}}$ are open in the topological space $(\B,\tau)$.
\end{engproof}

The topology generated by the subbase $P_{\mathcal{F}}$ is called \emph{topology generated by the compositional family} $\mathcal{F}$ and denoted by $\tau_{\mathcal{F}}$.

Proposition \ref{proposition2} and Lemma \ref{poligon_get_out} imply the following corollary.

\begin{engcorollary}\label{corollary2}
Every minimal semigroup topology on $\B$ is generated by some compositional family.
\end{engcorollary}

For arbitrary sets $A$ and $B$ we denote
\begin{align*}
    &A\subseteq^{*}B \mbox{~if a set~} A\setminus B \mbox{~is finite};\\
    &A=^{*}B \mbox{~if a set~} A\triangle B \mbox{~is finite}.\\
\end{align*}

\begin{engremark}
Note that there are semigroup topologies on $\B$ such that not generated by compositional families. In particular, a topology generated by the base
\begin{equation*}
  \mathcal{B}=\{\{(\alpha,\alpha)\mid\alpha\in A\}\cup\{0\}\mid A=^{*}\ab\}\cup\{\{(\alpha,\beta)\}\mid\alpha,\beta\in \ab\}.
\end{equation*}
\end{engremark}

\begin{engremark}
Observe that a semigroup topology on $\B$ can be generated by distinct compositional families. For example, for arbitrary sets $A\subset\ab$ and $B\subset A$ the following compositional families $\mathcal{F}_1=\{A\times(\ab\setminus A)\}$ and $\mathcal{F}_2=\{A\times(\ab\setminus A),B\times(\ab\setminus A)\}$ generate the same semigroup topology on $\B$.
\end{engremark}

Let $\tau$ be a semigroup topology on $\B$ generated by some compositional family. By $\C(\tau)$ denote the set of all compositional families such that generate the topology $\tau$.

\begin{engproposition}\label{proposition3}
Let $\tau_{1}$ and $\tau_{2}$ be semigroup topologies on $\B$ generated by compositional families. A topology $\tau_{1}$ is weaker than a topology $\tau_{2}$ if and only if there exist compositional families $\mathcal{F}_{1}\in\C(\tau_{1})$ and $\mathcal{F}_{2}\in\C(\tau_{2})$ such that $\mathcal{F}_{1}\subseteq \mathcal{F}_{2}$.
\end{engproposition}
\begin{engproof}
($\Rightarrow$) If $\mathcal{F}_{1}\in\C(\tau_{1})$ and $\mathcal{F}\in\C(\tau_{2})$, then the family $\mathcal{F}_{2}=\mathcal{F}_{1}\cup \mathcal{F}$ is compositional and $\mathcal{F}_{1}\subseteq \mathcal{F}_{2}$. Since the topology $\tau_{1}$ is weaker than the topology $\tau_{2}$, any element of $\mathcal{F}_{1}$ is a closed set in $(\B,\tau_{2})$. Therefore, the family $\mathcal{F}_{2}$ generate the topology $\tau_{2}$.

($\Leftarrow$) Since $\mathcal{F}_{1}\subseteq \mathcal{F}_{2}$, every closed set in $(\B,\tau_{1})$ is closed in $(\B,\tau_{2})$. Then the topology $\tau_{1}$ is weaker than the topology $\tau_{2}$.
\end{engproof}

\begin{englemma}\label{lemma5}
Let $\tau$ be a semigroup topology on $\B$. If the set $A\times B$ is closed in $(\B,\tau)$ and $C=^{*}A$, $D=^{*}B$, then the set $C\times D$ is closed in $(\B,\tau)$.
\end{englemma}
\begin{engproof}
By Lemma \ref{lemma1}, the set $(A\cap C)\times (B\cap D)$ is closed in $(\B,\tau)$. Lemma \ref{lemma2} implies that the sets $(A\cap C)\times \{\alpha\}$ and $\{\beta\}\times(B\cap D)$ are closed for all $\alpha\in D\setminus B, \beta\in C\setminus A$. Since the sets $D\setminus B$ and $C\setminus A$ are finite, the sets $(A\cap C)\times (D\setminus B)$ and $(C\setminus A)\times(B\cap D)$ are closed. The set $(C\setminus A)\times(D\setminus B)$ is finite and therefore closed. Hence the set $C\times D$ is closed in $(\B,\tau)$.
\end{engproof}

\section{Compositional digraphs}

A compositional family $\mathcal{F}$ can be represented in the form of a digraph with loops $D(\mathcal{F})$. The vertices of the digraph is the set
\begin{equation*}
  V(D(\mathcal{F}))=\{A\mid A\times(\ab\setminus B)\in\mathcal{F}\mbox{~or~}B\times(\ab\setminus A)\in \mathcal{F}\}
\end{equation*}
and $(A,B)\in A(D(\mathcal{F}))$ if $A\times(\ab\setminus B)\in\mathcal{F}$.

\begin{engdefinition}
A digraph with loops $D=(V,A)$ is called compositional if for all $(u,v)\in A$ there exists $w\in V$ such that $(u,w)\in A$ and $(w,v)\in A$.
\end{engdefinition}

\begin{engproposition}
For any compositional family $\mathcal{F}$ the digraph $D(\mathcal{F})$ is compositional and any compositional digraph $D$ such that vertices of $D$ are subsets of $\ab$ determines some compositional family.
\end{engproposition}

The following digraphs are compositional.

\begin{engproposition}\label{path}
There exists a hamiltonian path in $\mathcal{U}_{i}$ for each $i\in\mathbb{N}$.
\end{engproposition}
\begin{engproof}
Consider the sequence of vertices $W=0,\frac{1}{2^i},\frac{2}{2^i},\frac{3}{2^i},\ldots,\frac{2^i-1}{2^i},1$. Observe that $v_{j+1}-v_j=\frac{1}{2^i}$ for arbitrary $v_j, v_{j+1}\in W$. Thus $(v_j,v_{j+1})$ is an arc of $\mathcal{U}_{i}$ and hence this sequence is a path. Since the path contains all vertices of $\mathcal{U}_{i}$, $W$ is a hamiltonian path.
\end{engproof}

Proposition \ref{path} implies the following corollary.

\begin{engcorollary}\label{cycle}
Arbitrary quotient digraph of $\mathcal{U}$ has a finite cycle.
\end{engcorollary}

\begin{engproposition}
Let $\{D_{i}\}_{i\in I}$ be a collection of compositional digraphs. Arbitrary quotient digraph of $D=\bigoplus\limits_{i\in I}D_{i}$ is compositional.
\end{engproposition}
\begin{engproof}
Let $\mathcal{R}$ be an equivalence relation on $V(D)$ and $([u],[v])\in A(D/\mathcal{R})$. Then there exist $u',v'\in V(D)$ such that $(v',u')\in A(V)$. Thus $(v',u')$ is an arc of $V_j$ for some $j\in I$. Since digraph $V_j$ is compositional, there exist arcs $(u',w)$ and $(w,v')$ of $V_j$. Hence $([u'],[w]),([w],[v'])\in A(D/\mathcal{R})$.
\end{engproof}

\begin{englemma}\label{lemma6}
Let $D$ be a compositional digraph and $(u,v)$ be an arc of $D$. Then there exists an equivalence relation $\mathcal{R}$ on $V(\mathcal{U})$ such that $\mathcal{U}/\mathcal{R}$ is isomorphic to a subdigraph of $D$ which contains $(u,v)$.
\end{englemma}
\begin{engproof}
For each $w\in V(D)$ define a set $C_w$ of vertices of $\mathcal{U}$ by induction. For $i=0$ add vertex $0$ to $C_u$ and vertex $1$ to $C_v$.

Suppose that all vertexes of $\mathcal{U}_{n}$ are added to sets $\{C_w\}_{w\in V(D)}$. Take arbitrary vertex $y\in V(\mathcal{U}_{n+1})\setminus V(\mathcal{U}_{n})$. There exist two arcs $(x,y),(y,z)\in A(\mathcal{U}_{n+1})\setminus A(\mathcal{U}_{n})$ and $x\in C_r$, $z\in C_t$ for some $r,t\in V(D)$. Since the digraph $D$ is compositional, there exist arcs $(r,s),(s,t)\in A(D)$. Then add the vertex $y$ to $C_s$.

Nonempty elements of $\{C_w\}_{w\in V(D)}$ provide a partition of $V(\mathcal{U})$ which determines an equivalence relation $\mathcal{R}$. Let $H$ be an subdigraph of $D$ induced by $\{w\in V(D)\mid C_w\neq\varnothing\}$. Then the map $f:H\rightarrow\mathcal{U}/\mathcal{R}$ defined by the formula $f(w)=C_w$ is an isomorphism.
\end{engproof}

\begin{engproposition}\label{u_graph}
Any compositional digraph with size $\kappa$ is isomorphic to quotient digraph of $\bigoplus\limits_{k\in\kappa}\mathcal{U}$.
\end{engproposition}
\begin{engproof}
Let $D$ be compositional digraph with size $\kappa$. Consider a collection $\{\mathcal{U}^{(u,v)}\}_{(u,v)\in A(D)}$ where $\mathcal{U}^{(u,v)}$ is a digraph isomorphic to $\mathcal{U}$ for each $(u,v)\in A(D)$. By lemma \ref{lemma6}, for arbitrary $\mathcal{U}^{(u,v)}$ there exist an equivalence relation $\mathcal{R}_{(u,v)}$ and an isomorphism $f_{(u,v)}$ between $\mathcal{U}^{(u,v)}\diagup \mathcal{R}_{(u,v)}$ and subdigraph $H$ of $D$ which contains $(u,v)$.

Define the equivalence relation $\mathcal{R}$ on $\bigoplus\limits_{k\in\kappa}\mathcal{U}$ in the next way $u\mathcal{R}v$ if $f_{(s,t)}([u]_{\mathcal{R}_{(s,t)}})=f_{(x,y)}([v]_{\mathcal{R}_{(x,y)}})$ for $u\in V(\mathcal{U}^{(s,t)})$ and $v\in V(\mathcal{U}^{(x,y)})$. Now define the map $f:\left(\bigoplus\limits_{k\in\kappa}\mathcal{U}\right)/\mathcal{R}\rightarrow D$ defined by the formula $f([w])=f_{(s,t)}([w]_{\mathcal{R}_{(s,t)}})$ where $w\in\mathcal{U}^{(s,t)}$. Then $f$ is an isomorphism.
\end{engproof}

Corollary \ref{cycle} and Proposition \ref{u_graph} imply the following corollaries.

\begin{engcorollary}
Any compositional digraph has a finite cycle or contains $\mathcal{U}$ as subdigraph.
\end{engcorollary}

\begin{engcorollary}
Any finite compositional digraph has a finite cycle.
\end{engcorollary}

\section{Main result}

\begin{engproposition}\label{min_topology}
If $A\subset\ab$, then a topology $\tau$ generated by the compositional family $\{A\times(\ab\setminus A)\}$ is minimal.
\end{engproposition}
\begin{engproof}
Let $\tau_{1}$ be a weaker topology than the topology $\tau$ and $B\times(\ab\setminus C)$ be a closed set in $(\B,\tau_{1})$. By Lemma \ref{poligon_get_out}, there exists $D\subseteq\ab$ such that the sets $B\times(\ab\setminus D)$ and $D\times(\ab\setminus C)$ are closed in $(\B,\tau_{1})$. Since $\tau_{1}$ is weaker than $\tau$, $D\subseteq^{*}A$ and $(\ab\setminus D)\subseteq^{*}(\ab\setminus A)$. Therefore $D=^{*}A$. Again, by Lemma \ref{poligon_get_out}, there exists $F\subseteq\ab$ such that the sets $D\times(\ab\setminus F)$ and $F\times(\ab\setminus C)$ are closed in $(\B,\tau_{1})$. Hence $F\subseteq^{*}A$ and $(\ab\setminus F)\subseteq^{*}(\ab\setminus A)$ and then $F=^{*}A$. By Lemma \ref{lemma5}, the set $(A\times(\ab\setminus A)$ is closed in the topological space $(\B,\tau_{1})$.
\end{engproof}

Proposition \ref{min_topology} generalizes \cite[Theorem 5]{Pavlyk}.

\begin{engproposition}\label{tail}
Let $\mathcal{F}$ be a compositional family. If there exists a finite subdigraph $H$ of $D(\mathcal{F})$ which doesn't contain sink or source,
then there exists $A\subseteq\ab$ such that the set $A\times (\ab\setminus A)$ is closed in the topological space $(\B,\tau_{\mathcal{F}})$.
\end{engproposition}
\begin{engproof}
Let $V(D)=\{A_{1},\ldots ,A_{n}\}$ and $H$ doesn't contain sink. For each $A_{i}\in V(H)$ there exits $A_{j}\in V(H)$ such that $A_{i}\times(\ab\setminus A_{j})$. Observe that
\begin{equation*}
  \ab\setminus(A_{1}\cup\ldots\cup A_{n})=(\ab\setminus A_{1})\cap\ldots\cap (\ab\setminus A_{n}).
\end{equation*}
If $(\ab\setminus A_{1})\cap\ldots\cap (\ab\setminus A_{n})=\varnothing$, then $A_{1}\cup\ldots\cup A_{n}=\ab$ and hence, by Lemma \ref{lemma2}, for each $\alpha\in\ab$ the set $(\ab\setminus\{\alpha\})\times\{\alpha\}$ is closed in the topological space $(\B,\tau_{\mathcal{F}})$. Let $(\ab\setminus A_{1})\cap\ldots\cap (\ab\setminus A_{n})\neq\varnothing$. By Lemma \ref{lemma1}, the set $A_{i}\times((\ab\setminus A_{1})\cap\ldots\cap (\ab\setminus A_{n}))$ is closed for any $A_{i}\in V(H)$. Hence
\begin{equation*}
  A_{1}\cup\ldots\cup A_{n}\times \ab\setminus(A_{1}\cup\ldots\cup A_{n})=\bigcup\limits_{i=1}^{n}A_{i}\times((\ab\setminus A_{1})\cap\ldots\cap (\ab\setminus A_{n}))
\end{equation*}
is closed in the topological space $(\B,\tau_{\mathcal{F}})$. The case with source is proved similarly.
\end{engproof}

Propositions \ref{min_topology} and \ref{tail} imply the following corollary.

\begin{engcorollary}\label{fin_cycle}
If $D(\mathcal{F})$ has a finite cycle, then $\mathcal{F}$ is a singleton or generates a nonminimal topology.
\end{engcorollary}

Corollaries \ref{cycle} and \ref{fin_cycle} imply the following theorem.

\begin{engtheorem}
Let $\tau$ be a semigroup topology on $\B$ generated by compositional family $\mathcal{F}$ such that $D(\mathcal{F})$ doesn't contain subgraph isomorphic to $\mathcal{U}$. The topology $\tau$ is minimal if and only if $\tau$ is generated by singleton compositional family.
\end{engtheorem}

\begin{engproblem}
Is there a minimal semigroup topology on $\B$ generated by a composition family $\mathcal{F}$ such that $D(\mathcal{F})$ is isomorphic to $\mathcal{U}$?
\end{engproblem}
